function highlightBuildingUnderCenter() { const scene = viewer.scene; const canvas = scene.canvas; const center = new Cesium.Cartesian2( canvas.clientWidth / 2, canvas.clientHeight / 2 ); // On regarde TOUT ce qu'il y a sous le centre de l'écran const picks = scene.drillPick(center); if (!Cesium.defined(picks) || picks.length === 0) { return; } let feature = null; for (let i = 0; i < picks.length; i++) { const p = picks[i]; // On veut un bâtiment OSM (feature 3D Tiles du tileset osmBuildings) if ( p instanceof Cesium.Cesium3DTileFeature && p.primitive === osmBuildings ) { feature = p; break; } } if (!feature) { return; } // Remet l'ancien bâtiment en blanc if (lastHighlightedFeature) { lastHighlightedFeature.color = Cesium.Color.WHITE; } // Colore le nouveau bâtiment en ROUGE feature.color = Cesium.Color.RED.withAlpha(0.9); lastHighlightedFeature = feature; }
